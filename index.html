/** * /api/send-email.js * Blue Carpet — Postmark email route * * POST body: { email: string, videoUrl: string } * * Flow: * 1. Validate inputs * 2. Fetch the video from the Replicate CDN URL * 3. Check size against Postmark's 10MB total message limit * - Under limit → attach the video directly (base64) * - Over limit → send a branded HTML email with a download link instead * 4. POST to https://api.postmarkapp.com/email with correct headers * * Required env vars: * POSTMARK_SERVER_TOKEN — your Postmark server API token * EMAIL_FROM — verified sender signature, e.g. "Blue Carpet " * * Drop this file into your existing /api directory alongside predict.js * and register the route in your server the same way as /api/predict. */ // ── CONFIG ────────────────────────────────────────────────────────────────── const POSTMARK_API = 'https://api.postmarkapp.com/email'; const SERVER_TOKEN = process.env.POSTMARK_SERVER_TOKEN; const FROM_ADDRESS = process.env.EMAIL_FROM || 'Blue Carpet '; // Postmark's hard limit is 10 MB total *after* base64 encoding. // Base64 inflates by ~33 %, so a 7.5 MB video ≈ 10 MB encoded. // We keep headroom for HTML body + headers by capping at 7 MB raw. const MAX_ATTACH_BYTES = 7 * 1024 * 1024; // 7 MB raw // ── HANDLER ───────────────────────────────────────────────────────────────── module.exports = async function sendEmailHandler(req, res) { // Only accept POST if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed' }); } const { email, videoUrl } = req.body || {}; // ── 1. Validate inputs ──────────────────────────────────────────────────── if (!email || !email.includes('@')) { return res.status(400).json({ error: 'Valid email address required.' }); } if (!videoUrl || !videoUrl.startsWith('http')) { return res.status(400).json({ error: 'Valid video URL required.' }); } if (!SERVER_TOKEN) { console.error('[send-email] POSTMARK_SERVER_TOKEN env var is not set.'); return res.status(500).json({ error: 'Email service not configured.' }); } // ── 2. Fetch the video ──────────────────────────────────────────────────── let videoBuffer; try { const videoRes = await fetch(videoUrl, { method: 'GET' }); if (!videoRes.ok) throw new Error(`Video fetch failed: ${videoRes.status}`); const arrayBuf = await videoRes.arrayBuffer(); videoBuffer = Buffer.from(arrayBuf); } catch (err) { console.error('[send-email] Could not fetch video:', err.message); return res.status(502).json({ error: 'Could not retrieve video file. Please try again.' }); } // ── 3. Build the Postmark payload ───────────────────────────────────────── const isAttachable = videoBuffer.byteLength <= MAX_ATTACH_BYTES; const videoBase64 = isAttachable ? videoBuffer.toString('base64') : null; const htmlBody = buildHtmlEmail(email, videoUrl, isAttachable); const textBody = buildTextEmail(email, videoUrl, isAttachable); const payload = { From: FROM_ADDRESS, To: email, Subject: 'Your Blue Carpet Moment — iS Clinical Future of Aesthetics 2026', HtmlBody: htmlBody, TextBody: textBody, MessageStream: 'outbound', Tag: 'blue-carpet-video', TrackOpens: true, }; // Attach video if small enough, otherwise the email body has the download link if (isAttachable && videoBase64) { payload.Attachments = [ { Name: 'blue-carpet-moment.mp4', Content: videoBase64, ContentType: 'video/mp4', }, ]; } // ── 4. POST to Postmark ─────────────────────────────────────────────────── let pmRes; try { pmRes = await fetch(POSTMARK_API, { method: 'POST', headers: { 'Accept': 'application/json', 'Content-Type': 'application/json', 'X-Postmark-Server-Token': SERVER_TOKEN, }, body: JSON.stringify(payload), }); } catch (err) { console.error('[send-email] Postmark request failed:', err.message); return res.status(502).json({ error: 'Failed to reach email service. Please try again.' }); } const pmData = await pmRes.json(); // Postmark returns ErrorCode 0 for success if (!pmRes.ok || pmData.ErrorCode !== 0) { console.error('[send-email] Postmark error:', pmData); return res.status(502).json({ error: pmData.Message || 'Email delivery failed. Please try again.', }); } console.log(`[send-email] Sent to ${email} — MessageID: ${pmData.MessageID} — attached: ${isAttachable}`); return res.status(200).json({ success: true, messageId: pmData.MessageID, attached: isAttachable, }); }; // ── EMAIL TEMPLATES ────────────────────────────────────────────────────────── function buildHtmlEmail(email, videoUrl, attached) { const videoBlock = attached ? `

Your video is attached to this email as blue-carpet-moment.mp4.

` : `

Your video is ready to download — tap the button below.

Download Your Moment `; return ` Your Blue Carpet Moment

iS Clinical · Future of Aesthetics 2026

Blue Carpet

Your moment is ready

You stepped onto the blue carpet.
Here's the cinematic result.

${videoBlock}

Generated exclusively at the iS Clinical Future of Aesthetics 2026 event.
Share your moment and tag us.

iS Clinical Skincare South Africa · sternlaser.co.za

`; } function buildTextEmail(email, videoUrl, attached) { const videoLine = attached ? 'Your blue carpet video is attached to this email as blue-carpet-moment.mp4.' : `Your blue carpet video is ready to download:\n${videoUrl}`; return `YOUR BLUE CARPET MOMENT iS Clinical · Future of Aesthetics 2026 You stepped onto the blue carpet. Here's your cinematic moment. ${videoLine} Generated exclusively at the iS Clinical Future of Aesthetics 2026 event. iS Clinical Skincare South Africa · sternlaser.co.za`; }
